\section{Phản ứng Oxi hóa - khử}
\begin{Muctieu}
	\begin{itemize}
		\item Nêu được khái niệm và xác định được số oxi hoá của nguyên tử các nguyên tố trong hợp chất.
		\item Nêu được khái niệm và ý nghĩa của phản ứng oxi hoá - khử.
		\item Lập được phương trình hoá học của phản ứng oxi hoá - khử bằng phương pháp thăng bằng electron.
		\item Mô tả được một số phản ứng oxi hoá - khử quan trọng trong cuộc sống.
	\end{itemize}
\end{Muctieu}
