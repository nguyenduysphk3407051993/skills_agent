% Macro để vẽ nhiều nguyên tử
\newcommand{\drawarrangement}[2][]{% #1: options, #2: list of atoms
	\begin{tikzpicture}[
		declare function={
			r1=0.5cm;
			r2=1.0cm;
			r2h=r2+4pt;
			r3=r2+1.0cm;
			r3h=r3+6pt;
			r4=r3+1.3cm;
			r4h=r4+6pt;
			d=4cm;  % Khoảng cách giữa các nguyên tử
		},
		stylefont/.style={font=\bfseries\large\fontfamily{qag}\selectfont,text=\mycolor!25!black},
		#1
		]
		\foreach \element [count=\i] in {#2} {
			\path (\i*d-d,0) pic[local bounding box=atom\i] {atom={r1/r2/r3/r4/\element/\mycolor/\element}};
			\path ($(atom\i.center)+(-90:r4+0.5cm)$) 
			node[text width=3cm,align=center,anchor=north,stylefont] 
			{\element\\Nguyên tử \element};
		}
	\end{tikzpicture}
}


% Hoặc sắp xếp thành lưới:
\newcommand{\drawatomgrid}[3]{% #1: rows, #2: cols, #3: list of atoms
	\begin{tikzpicture}[
		declare function={
			r1=0.5cm;
			r2=1.0cm;
			r2h=r2+4pt;
			r3=r2+1.0cm;
			r3h=r3+6pt;
			r4=r3+1.3cm;
			r4h=r4+6pt;
			d=4cm;  % Khoảng cách ngang
			v=5cm;  % Khoảng cách dọc
		},
		stylefont/.style={font=\bfseries\large\fontfamily{qag}\selectfont,text=\mycolor!25!black}
		]
		\foreach \element [count=\i] in {#3} {
			\pgfmathsetmacro{\row}{int((\i-1)/#2)}
			\pgfmathsetmacro{\col}{int(mod(\i-1,#2))}
			\path (\col*d,-\row*v) pic[local bounding box=atom\i] {atom={r1/r2/r3/r4/\element/\mycolor/\element}};
			\path ($(atom\i.center)+(-90:r4+0.5cm)$) 
			node[text width=3cm,align=center,anchor=north,stylefont] 
			{\element\\Nguyên tử \element};
		}
	\end{tikzpicture}
}
